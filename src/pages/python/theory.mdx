def main(m, a, n):
	sumn = 0

	for i in range(1, n):
		proda = 1

		for j in range(1, a):
			summ = 0

			for k in range(1, m):
				summ += (22 * i ** 3 - 9 * j - k ** 2) - j ** 5

			proda *= summ
		sumn += proda
	return sumn
